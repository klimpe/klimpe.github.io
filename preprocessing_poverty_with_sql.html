Poverty Data Extraction

          
SELECT *
FROM `drugs-and-poverty.drugs_and_poverty.poverty_1` AS out
INNER JOIN (
    SELECT country, MAX(year) AS max_year
    FROM `drugs-and-poverty.drugs_and_poverty.poverty_1`
    GROUP BY
      country
    ) AS sub
  ON out.country = sub.country AND out.year = sub.max_year
WHERE ppp_version = 2017
ORDER BY out.country
